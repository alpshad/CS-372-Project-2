Func calcMultiples x, y, m
    count = 0
    index = 0
    While x * index <= m Do
        count = count + 1
        index = index + 1
    End

    //Say "After first while loop"
    index = 0
    While y * index <= m Do
        count = count + 1
        index = index + 1
    End

    Say "Hello"
    Say count
End
